// ================= CERTIFICATE FILE VERIFICATION & AUTO-REGENERATION =================
// Backend: code.gs
// This file contains all functions needed to verify certificate file existence
// and automatically regenerate missing certificates on demand
// ==================================================================================

/**
 * CHECK IF CERTIFICATE FILE EXISTS IN GOOGLE DRIVE
 * This function verifies if a PDF file is still accessible at the given URL
 * 
 * @param {string} pdfUrl - The URL of the certificate PDF
 * @param {string} certId - Certificate ID for logging
 * @returns {object} { exists: boolean, isAccessible: boolean, fileId: string|null }
 */
function verifyCertificateFileExists_(pdfUrl, certId) {
  try {
    if (!pdfUrl || String(pdfUrl).trim() === '') {
      return { exists: false, isAccessible: false, fileId: null, reason: 'Empty URL' };
    }

    // Extract file ID from URL (Google Drive URLs contain /d/[FILE_ID]/ or id=[FILE_ID])
    let fileId = null;
    
    // Pattern 1: /d/FILE_ID/view or /d/FILE_ID/
    const match1 = pdfUrl.match(/\/d\/([a-zA-Z0-9-_]+)/);
    if (match1) {
      fileId = match1[1];
    }
    // Pattern 2: id=FILE_ID
    if (!fileId) {
      const match2 = pdfUrl.match(/[?&]id=([a-zA-Z0-9-_]+)/);
      if (match2) {
        fileId = match2[1];
      }
    }

    if (!fileId) {
      Logger.log('Certificate ' + certId + ': Could not extract file ID from URL: ' + pdfUrl);
      return { exists: false, isAccessible: false, fileId: null, reason: 'Invalid URL format' };
    }

    try {
      // Attempt to access the file
      const file = DriveApp.getFileById(fileId);
      
      // Check if file exists and is still valid
      if (file && !file.isTrashed()) {
        Logger.log('Certificate ' + certId + ': File exists and is accessible');
        return { 
          exists: true, 
          isAccessible: true, 
          fileId: fileId,
          fileName: file.getName(),
          fileSize: file.getSize()
        };
      } else if (file && file.isTrashed()) {
        Logger.log('Certificate ' + certId + ': File exists but is in trash');
        return { exists: false, isAccessible: false, fileId: fileId, reason: 'File is in trash' };
      }
    } catch(fileError) {
      Logger.log('Certificate ' + certId + ': File not found or access denied: ' + fileError.toString());
      return { exists: false, isAccessible: false, fileId: fileId, reason: 'Access denied or not found' };
    }

  } catch(error) {
    Logger.log('Certificate verification error: ' + error.toString());
    return { exists: false, isAccessible: false, fileId: null, reason: error.toString() };
  }

  return { exists: false, isAccessible: false, fileId: null, reason: 'Unknown error' };
}

/**
 * REGENERATE CERTIFICATE IF MISSING
 * If certificate file is missing, this function regenerates it from the template
 * 
 * @param {object} sheet - SpreadSheet object
 * @param {string} certId - Certificate ID to regenerate
 * @returns {object} { status: 'success'|'error', message: string, pdf_url: string|null }
 */
function regenerateMissingCertificate_(sheet, certId) {
  try {
    const wsCert = getCertificateRegistrySheet_(sheet);
    const wsEnroll = sheet.getSheetByName('Enrollments');
    const wsCourse = getCoursesSheet_(sheet);

    // ===== STEP 1: GET CERTIFICATE RECORD =====
    const certRecord = findCertificateRecordById_(wsCert, certId);
    if (!certRecord || !certRecord.rowIndex) {
      return { 
        status: 'error', 
        message: 'ไม่พบข้อมูลใบประกาศ (Certificate ID: ' + certId + ')',
        pdf_url: null
      };
    }

    Logger.log('Regenerating certificate: ' + certId);

    // ===== STEP 2: GET COURSE TEMPLATE INFO =====
    const courseData = wsCourse.getDataRange().getValues();
    let templateId = '';
    let courseTitle = '';
    
    for (let i = 1; i < courseData.length; i++) {
      if (courseData[i][0] === certRecord.courseId) {
        const course = buildCourseRecord_(courseData[i]);
        templateId = course.cert_template;
        courseTitle = course.title;
        break;
      }
    }

    if (!templateId) {
      return { 
        status: 'error', 
        message: 'ไม่พบ Template สำหรับหลักสูตรนี้',
        pdf_url: null
      };
    }

    // ===== STEP 3: GET TEMPLATE FILE =====
    let templateFile;
    try {
      templateFile = DriveApp.getFileById(templateId);
    } catch(e) {
      return { 
        status: 'error', 
        message: 'ไม่สามารถเข้าถึง Template ใบประกาศได้',
        pdf_url: null
      };
    }

    // ===== STEP 4: SETUP OUTPUT FOLDER =====
    let destFolder;
    const CERT_FOLDER_ID = '19Htb9AjawXDOzje5yVeyT_i7mkYh-k45';
    try {
      destFolder = DriveApp.getFolderById(CERT_FOLDER_ID);
    } catch(fe) {
      const certFolders = DriveApp.getFoldersByName('E-Learning Certificates');
      destFolder = certFolders.hasNext() ? certFolders.next() : DriveApp.createFolder('E-Learning Certificates');
    }

    // ===== STEP 5: CREATE PRESENTATION FROM TEMPLATE =====
    const certFileName = 'Cert_' + certId + '_' + sanitizeFileName_(certRecord.userName) + '_' + sanitizeFileName_(courseTitle);
    const tempFile = templateFile.makeCopy(certFileName, destFolder);
    const tempPresentation = SlidesApp.openById(tempFile.getId());
    const slide = tempPresentation.getSlides()[0];

    // ===== STEP 6: REPLACE TEXT PLACEHOLDERS =====
    const today = new Date().toLocaleDateString('th-TH', { 
      year: 'numeric', 
      month: 'long', 
      day: 'numeric' 
    });

    try {
      slide.replaceAllText('{{NAME}}', certRecord.userName);
      slide.replaceAllText('{{COURSE}}', courseTitle);
      slide.replaceAllText('{{CAUSE}}', courseTitle);
      slide.replaceAllText('{{DATE}}', today);
      slide.replaceAllText('{{CERT_ID}}', certId);
      slide.replaceAllText('{{VERIFY_URL}}', certRecord.verifyUrl);
    } catch(re) {
      Logger.log('Text replacement warning during regeneration: ' + re.toString());
    }

    // ===== STEP 7: INSERT QR CODE =====
    try {
      insertQrCodePlaceholders_(tempPresentation, certRecord.verifyUrl);
    } catch(qe) {
      Logger.log('QR code insertion skipped during regeneration: ' + qe.toString());
    }

    // ===== STEP 8: SAVE & CONVERT TO PDF =====
    tempPresentation.saveAndClose();
    const pdfBlob = tempFile.getAs('application/pdf').setName(certId + '.pdf');
    const pdfFile = destFolder.createFile(pdfBlob);
    pdfFile.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);

    // ===== STEP 9: CLEANUP & UPDATE RECORD =====
    tempFile.setTrashed(true);

    const newPdfUrl = pdfFile.getUrl();
    
    // Update certificate registry
    wsCert.getRange(certRecord.rowIndex, 8).setValue(newPdfUrl);

    // Update enrollment record with new PDF URL
    const enrollData = wsEnroll.getDataRange().getValues();
    for (let i = 1; i < enrollData.length; i++) {
      if (enrollData[i][1] === certRecord.userId && enrollData[i][2] === certRecord.courseId) {
        wsEnroll.getRange(i + 1, 8).setValue(newPdfUrl);
        break;
      }
    }

    Logger.log('Certificate regenerated successfully: ' + certId);
    
    return { 
      status: 'success', 
      message: 'ใบประกาศถูกสร้างใหม่เรียบร้อยแล้ว',
      pdf_url: newPdfUrl,
      cert_id: certId
    };

  } catch(error) {
    Logger.log('Certificate regeneration error: ' + error.toString());
    return { 
      status: 'error', 
      message: 'ไม่สามารถสร้างใบประกาศใหม่ได้: ' + error.toString(),
      pdf_url: null
    };
  }
}

/**
 * MAIN API ENDPOINT - VERIFY AND DOWNLOAD CERTIFICATE
 * Called from frontend when user clicks download button
 * Checks if file exists, regenerates if missing, then returns download URL
 * 
 * @param {object} sheet - SpreadSheet object
 * @param {object} payload - { user_id, course_id, cert_id (optional) }
 * @returns {object} { status, message, pdf_url, cert_id }
 */
function verifyCertificateAndDownload(sheet, payload) {
  try {
    const wsCert = getCertificateRegistrySheet_(sheet);
    const wsEnroll = sheet.getSheetByName('Enrollments');

    // ===== STEP 1: FIND CERTIFICATE RECORD =====
    let certRecord = null;

    if (payload.cert_id) {
      // Search by certificate ID
      certRecord = findCertificateRecordById_(wsCert, payload.cert_id);
    } else {
      // Search by user + course
      certRecord = findCertificateRecordByUserCourse_(wsCert, payload.user_id, payload.course_id);
    }

    if (!certRecord) {
      return responseJSON({ 
        status: 'error', 
        message: 'ไม่พบข้อมูลใบประกาศสำหรับผู้ใช้นี้' 
      });
    }

    Logger.log('Certificate download requested: ' + certRecord.certId);

    // ===== STEP 2: VERIFY FILE EXISTS =====
    const fileCheck = verifyCertificateFileExists_(certRecord.pdfUrl, certRecord.certId);
    
    if (fileCheck.exists && fileCheck.isAccessible) {
      Logger.log('Certificate file verified: ' + certRecord.certId);
      // File exists and is accessible - return URL directly
      return responseJSON({ 
        status: 'success', 
        message: 'ใบประกาศพร้อมดาวน์โหลด',
        pdf_url: certRecord.pdfUrl,
        cert_id: certRecord.certId,
        verify_url: certRecord.verifyUrl,
        file_verified: true
      });
    }

    // ===== STEP 3: FILE MISSING - REGENERATE AUTOMATICALLY =====
    Logger.log('Certificate file missing or inaccessible, regenerating: ' + certRecord.certId);
    Logger.log('Reason: ' + fileCheck.reason);

    const regenerateResult = regenerateMissingCertificate_(sheet, certRecord.certId);

    if (regenerateResult.status === 'success') {
      Logger.log('Certificate regenerated successfully, returning new URL');
      return responseJSON({ 
        status: 'success', 
        message: 'ใบประกาศถูกสร้างใหม่เรียบร้อย',
        pdf_url: regenerateResult.pdf_url,
        cert_id: certRecord.certId,
        verify_url: certRecord.verifyUrl,
        file_regenerated: true
      });
    } else {
      Logger.log('Certificate regeneration failed: ' + regenerateResult.message);
      return responseJSON({ 
        status: 'error', 
        message: 'ไม่สามารถสร้างใบประกาศใหม่ได้: ' + regenerateResult.message 
      });
    }

  } catch(error) {
    Logger.log('Certificate verification/download error: ' + error.toString());
    return responseJSON({ 
      status: 'error', 
      message: 'เกิดข้อผิดพลาดระหว่างตรวจสอบใบประกาศ: ' + error.toString() 
    });
  }
}

/**
 * HELPER: Sanitize filename
 */
function sanitizeFileName_(name) {
  return String(name)
    .replace(/[\/\\?*:|"<>]/g, '')
    .substring(0, 50);
}
